import HorsesDto from './Horses.dto.js';
/**
 *
 * @param {*} req
 * @param {*} res
 * @returns {Promise<*>}
 * @description Obtener todos los equinos
 */

const getAllHorses = async (req, res) => {
	try {
		const datos = req.query;
		console.log(req.query);
		const { statusCode, data } = await HorsesDto.getAllHorses();
		res.status(statusCode).json(data);
	} catch (error) {
		throw error;
	}
};

const getAllHorsesWithPaginations = async (req, res) => {
	try {
		const { _limit, _offset } = req.query;
		const { statusCode, data } = await HorsesDto.getAllHorsesWithPaginations(
			_limit,
			_offset
		);
		res.status(statusCode).json(data);
	} catch (error) {
		throw error;
	}
};

export default {
	getAllHorses,
	getAllHorsesWithPaginations
};
